create or replace FUNCTION FN_GET_CODE_NM ( P_MASTER_CODE IN VARCHAR2,P_CODE IN VARCHAR2 ) RETURN VARCHAR2 IS
/*
 * 함 수 명 : FN_GET_CODE_NM
 * 설    명 : 코드명을 가져 오는 함수
*/
RET_VALUE    RBS_OPTION_INFO.OPTION_NAME%TYPE;
BEGIN
    SELECT OPTION_NAME
        INTO RET_VALUE
    FROM RBS_OPTION_INFO
    WHERE MASTER_CODE = P_MASTER_CODE
      AND OPTION_CODE   = P_CODE ;

    RETURN  RET_VALUE ;
EXCEPTION WHEN OTHERS THEN
    RETURN '';
END;
