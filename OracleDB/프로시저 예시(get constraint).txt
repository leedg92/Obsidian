create or replace FUNCTION FN_GET_CONSTRAINT(I_TABLE_NAME IN VARCHAR2, I_COLUMN_NAME IN VARCHAR2) RETURN VARCHAR2 IS

iCNT1   INTEGER;
iCNT2   INTEGER;
BEGIN
    SELECT COUNT(*) INTO iCNT1
    FROM USER_CONSTRAINTS A
     JOIN USER_CONS_COLUMNS B ON ( A.TABLE_NAME = B.TABLE_NAME AND A.CONSTRAINT_NAME = B.CONSTRAINT_NAME )
    WHERE A.TABLE_NAME =I_TABLE_NAME
       AND CONSTRAINT_TYPE = 'P' 
       AND B.COLUMN_NAME = I_COLUMN_NAME ;

    SELECT COUNT(*) INTO iCNT2
    FROM USER_CONSTRAINTS A
     JOIN USER_CONS_COLUMNS B ON ( A.TABLE_NAME = B.TABLE_NAME AND A.CONSTRAINT_NAME = B.CONSTRAINT_NAME )
    WHERE A.TABLE_NAME =I_TABLE_NAME
       AND CONSTRAINT_TYPE = 'R' 
       AND B.COLUMN_NAME = I_COLUMN_NAME ;

    IF iCNT1 > 0 AND iCNT2 > 0 THEN
        RETURN 'PFK';
    ELSIF iCNT1 > 0 AND iCNT2 = 0 THEN
        RETURN 'PK';
    ELSIF iCNT1 = 0 AND iCNT2 > 0 THEN
        RETURN 'FK';
    ELSE
        RETURN '';
    END IF;

END;
