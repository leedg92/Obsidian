
--대학
SELECT 
 DEPT_CD,
 DEPT_KOR_NM 
FROM 
 COM.CMMN050
WHERE 
 POSI_GRP_CD = 'C0008021' /*대학*/
 AND USE_FG = 1;


--학부/학과
SELECT 
 DEPT_CD,
 DEPT_KOR_NM 
FROM 
 COM.CMMN050
WHERE DEPT_CD LIKE '선택학 대학코드%'
 AND DEPT_CD != '선택한 대학코드'
 AND POSI_GRP_CD = 'C0008022' /*학부*/
 AND USE_FG = 1;

--전공
SELECT 
 DEPT_CD,
 DEPT_KOR_NM 
FROM 
 COM.CMMN050
WHERE DEPT_CD LIKE '선택한 학부/학과코드%'
 AND DEPT_CD != '선택한 학부/학과코드'
 AND POSI_GRP_CD = 'C0008023' /*전공*/
 AND USE_FG = 1;
