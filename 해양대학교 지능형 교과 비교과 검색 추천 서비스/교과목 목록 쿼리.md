- ### 교과목 목록
	```
		SELECT  
              B.SUBJECT_CD  
            ,  A.SUBJECT_NM  
            ,  A.SUBJECT_ENM  
           ,  B.DEPT_CD  
            ,  (SELECT DEPT_NM FROM SCO.V_AI_ORG WHERE DEPT_CD = B.DEPT_CD) AS DEPT_NAME  
           ,  B.YEAR  
           ,  B.COM_GRADE  
           ,  B.SMT  
            ,  UF_GET_SYS020TB(B.SMT) AS SMT_NAME  
            ,  B.COMDIV_CODE  
            ,  UF_GET_SYS020TB(B.COMDIV_CODE) AS COMDIV_CODE_NAME  
            ,  B.MNRCOM_DIV_CODE  
            ,  UF_GET_SYS020TB(B.MNRCOM_DIV_CODE) AS MNRCOM_DIV_CODE_NAME  
            , B.FUSE_DEPT_CD  
            , (SELECT DEPT_NM FROM SCO.V_AI_ORG WHERE DEPT_CD = B.FUSE_DEPT_CD) AS FUSE_DEPT_CD_NAME
       FROM  
          SUF.SUF010TB A,  
          SUF.SUF050TB B  
       WHERE  
          A.ABO_DT IS NULL           --폐기일자 NULL  
       AND A.SINBUN_CODE = 'UD340001' --학부생만  
       AND A.SUBJECT_CD = B.SUBJECT_CD  
       AND A.FIR_YEAR = B.YEAR  
       AND A.FIR_SMT = B.SMT  
        AND SUBJECT_NM LIKE '%%'  
        AND B.DEPT_CD = '';


	```
