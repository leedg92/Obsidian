                 AND DECODE(A.NEW_PASS_YN,'Y',A.CUBE_PASSWD,A.PASSWD) = DECODE(A.NEW_PASS_YN,'Y',XX1.ENC_VARCHAR2_INS(as_USER_PW,10,'PASSWD','SYS111TB','CUBE_PASSWD'),HASH_SHA512(as_USER_PW))
